        async function getCurrentTrack() {
            try {
                const url = `https://ws.audioscrobbler.com/2.0/?method=user.getRecentTracks&user=${LASTFM_USERNAME}&api_key=${LASTFM_API_KEY}&format=json&limit=1`;
                const proxyUrl = `https://api.allorigins.win/raw?url=${encodeURIComponent(url)}`;
                
                debugLog(`Начинаем запрос к Last.fM API...`);
                
                let response = await fetchWithTimeout(url, 5000).catch(() => null);
                
                if (!response || !response.ok) {
                    debugLog(`Прямой запрос не удался, пробуем CORS proxy...`);
                    response = await fetchWithTimeout(proxyUrl, 5000).catch(() => null);
                }
                
                if (response && response.ok) {
                    const data = await response.json();
                    debugLog(`Данные успешно получены!`);
                    
                    if (data.recenttracks && data.recenttracks.track.length > 0) {
                        const track = data.recenttracks.track[0];
                        const isNowPlaying = track['@attr'] && track['@attr']['nowplaying'] === 'true';
                        updatePlayer(track, isNowPlaying);
                        return true;
                    } else {
                        debugLog(`Нет данных о треках в ответе`);
                        showNoTracks();
                    }
                } else {
                    throw new Error('Все методы запроса не удались');
                }
            } catch (error) {
                console.error('Error fetching Last.fM data:', error);
                debugLog(`Ошибка: ${error.message}`);
                showError();
            }
            return false;
        }

        function fetchWithTimeout(url, timeout = 5000) {
            return Promise.race([
                fetch(url),
                new Promise((_, reject) =>
                    setTimeout(() => reject(new Error('Timeout')), timeout)
                )
            ]);
        }

        function updatePlayer(track, isNowPlaying) {
            const noTrack = document.getElementById('no-track');
            const trackContent = document.getElementById('track-content');
            const status = document.getElementById('player-status');
            
            noTrack.style.display = 'none';
            trackContent.style.display = 'block';
            
            document.getElementById('track-name').textContent = track.name || 'Неизвестный трек';
            document.getElementById('artist-name').textContent = track.artist['#text'] || 'Неизвестный исполнитель';
            document.getElementById('album-name').textContent = track.album['#text'] || 'Неизвестный альбом';
            
            const albumArt = document.getElementById('album-art');
            const imageUrl = track.image && track.image[2] && track.image[2]['#text'];
            
            if (imageUrl) {
                const img = new Image();
                img.onload = function() {
                    albumArt.innerHTML = '';
                    const newImg = document.createElement('img');
                    newImg.src = imageUrl;
                    newImg.alt = "Album Art";
                    albumArt.appendChild(newImg);
                };
                img.onerror = function() {
                    albumArt.innerHTML = '<i class="fas fa-music"></i>';
                };
                img.src = imageUrl;
            } else {
                albumArt.innerHTML = '<i class="fas fa-music"></i>';
            }
            
            if (isNowPlaying) {
                status.textContent = '🎵 СЕЙЧАС ИГРАЕТ';
                status.style.background = 'rgba(255, 59, 109, 0.3)';
                isPlaying = true;
                document.getElementById('play-icon').className = 'fas fa-pause';
                startVisualizer();
            } else {
                status.textContent = '⏸️ НЕАКТИВЕН';
                status.style.background = 'rgba(128, 128, 128, 0.3)';
                isPlaying = false;
                document.getElementById('play-icon').className = 'fas fa-play';
                stopVisualizer();
            }
            
            updateLastUpdated();
        }

            stopVisualizer();

            if (visualizerInterval) {
                clearInterval(visualizerInterval);
                const bars = document.querySelectorAll('.bar');
                bars.forEach(bar => bar.style.height = '5px');
            }
        }

            "Что-то между прогером и скрипт-кидди",

        const typingElement = document.getElementById('typing-text');
        let currentPhrase = 0;
        let charIndex = 0;
        let isDeleting = false;
        let typingSpeed = 100;
        
        function type() {
            const fullText = phrases[currentPhrase];
            
            if (isDeleting) {
                typingElement.textContent = fullText.substring(0, charIndex - 1);
                charIndex--;
                typingSpeed = 50;
            } else {
                typingElement.textContent = fullText.substring(0, charIndex + 1);
                charIndex++;
                typingSpeed = 100 + Math.random() * 50;
            }
            
            if (!isDeleting && charIndex === fullText.length) {
                isDeleting = true;
                typingSpeed = 1500;
            } else if (isDeleting && charIndex === 0) {
                isDeleting = false;
                currentPhrase = (currentPhrase + 1) % phrases.length;
                typingSpeed = 500;
            }
            
            setTimeout(type, typingSpeed);
        }
        
        window.addEventListener('load', function() {
            setTimeout(type, 1500);
        });
